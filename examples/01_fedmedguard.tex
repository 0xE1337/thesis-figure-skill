\begin{tikzpicture}[

    every node/.style={font=\footnotesize},
    base_box/.style={
        rectangle, rounded corners=3pt, align=center,
        minimum height=0.9cm, minimum width=2.6cm,
        drop shadow={opacity=0.15}, thick
    },
    blue_node/.style={base_box, fill=drawBlueFill, draw=drawBlueLine},
    green_node/.style={base_box, fill=drawGreenFill, draw=drawGreenLine},
    orange_node/.style={base_box, fill=drawOrangeFill, draw=drawOrangeLine},
    purple_node/.style={base_box, fill=drawPurpleFill, draw=drawPurpleLine},
    grey_node/.style={base_box, fill=drawGreyFill, draw=drawGreyLine},
    arrow/.style={-{Stealth[scale=1.2]}, thick, color=black!70},
    arrow_data/.style={-{Stealth[scale=1.2]}, very thick, color=drawOrangeLine},
    arrow_dash/.style={-{Stealth[scale=1.2]}, dashed, thick, color=black!70},
    zone/.style={dashed, thick, inner sep=15pt, rounded corners=8pt},
    layer_title/.style={font=\bfseries\small, inner sep=5pt}
]

% ===== 底层：本地训练层 =====
\node[blue_node] (hospA) {医院节点 A};
\node[blue_node, right=1.8cm of hospA] (hospB) {医院节点 B};
\node[blue_node, right=1.8cm of hospB] (hospC) {医院节点 C};

% ===== 中层：安全聚合层 =====
\node[green_node, above=2.2cm of hospA] (dp) {差分隐私注入};
\node[green_node, right=1.8cm of dp] (smpc) {安全多方计算};
\node[orange_node, right=1.8cm of smpc] (agg) {全局模型聚合};

% ===== 顶层：模型评估与分发层 =====
\node[purple_node, above=2.2cm of smpc] (eval) {联邦评估};
\node[purple_node, right=1.8cm of eval] (version) {模型版本管理};

% ===== 独立模块：审计日志 =====
\node[grey_node, right=2.2cm of agg] (audit) {审计日志};

% ===== 连线 =====
% 底层 → 中层：模型梯度上传
\draw[arrow] (hospA.north) -- ++(0,0.7) -| (dp.south);
\draw[arrow] (hospB.north) -- ++(0,0.7) -| (dp.south east);
\draw[arrow] (hospC.north) -- ++(0,0.7) -| (smpc.south);
\node[font=\footnotesize, anchor=west] at ($(hospB.north)+(0.15,0.35)$) {模型梯度};

% 中层内部连线
\draw[arrow] (dp) -- node[above, font=\footnotesize] {加噪梯度} (smpc);
\draw[arrow] (smpc) -- (agg);

% 中层 → 顶层：全局参数
\coordinate (mid_agg_eval) at ($(agg.north)+(0,0.7)$);
\draw[arrow_data] (agg.north) -- (mid_agg_eval) -| (eval.south east);
\node[font=\footnotesize, anchor=west] at ($(mid_agg_eval)+(0.1,0)$) {全局参数};

% 顶层内部连线
\draw[arrow] (eval) -- (version);

% 回路：全局模型下发（从右侧绕行向下到所有医院节点）
\coordinate (loop_top) at ($(version.east)+(0.6,0)$);
\coordinate (loop_bot) at ($(hospC.south)+(0,-0.6)$);
\coordinate (loop_left) at ($(hospA.south)+(-0.6,-0.6)$);

\draw[arrow_dash] (version.east) -- (loop_top) -- (loop_top|-loop_bot) -- (loop_left|-loop_bot) -- (loop_left|-hospA.south) -- (hospA.south);
\draw[arrow_dash] (loop_left|-loop_bot) -| (hospB.south);
\draw[arrow_dash] (loop_left|-loop_bot) -| (hospC.south);
\node[font=\footnotesize, anchor=east] at ($(loop_top|-loop_bot)+(-0.1,0.3)$) {加密下发全局模型};

% 虚线：到审计日志
\draw[arrow_dash] (hospC.east) -- ++(0.3,0) |- (audit.south west);
\node[font=\footnotesize, anchor=south] at ($(hospC.east)+(1.2,0.15)$) {上传记录};

\draw[arrow_dash] (version.east) -- ++(0.3,0) |- (audit.north west);
\node[font=\footnotesize, anchor=south] at ($(version.east)+(1.0,0.15)$) {下发记录};
